function runNTT() { if (!runtimeInitialized) { updateStatus('❌ Bitte zuerst Laufzeit initialisieren', 'error'); return; } updateStatus('⚡ Führe NTT 4096 aus (echtes WASM)...', 'loading'); runNttBtn.disabled = true; // WICHTIG: Echte WASM-Funktion aufrufen if (window.wasmExports && typeof window.wasmExports.ntt_4096 === 'function') { try { const startTime = performance.now(); // ✅ DIESER AUFRUF RUFT DIE RUST-FUNKTION AUF! const result = window.wasmExports.ntt_4096(); const endTime = performance.now(); const time = (endTime - startTime).toFixed(1); updateStatus(`✅ NTT 4096 abgeschlossen in ${time} ms`, 'success'); addResult(`NTT 4096 abgeschlossen in ${time} ms | Ergebnis: ${result}`); } catch (error) { updateStatus(`❌ Fehler bei NTT: ${error.message}`, 'error'); addResult(`NTT Fehler: ${error.message}`, true); } } else { updateStatus('❌ NTT-Funktion nicht in WASM gefunden', 'error'); addResult('Die NTT-Funktion wurde im WASM-Code nicht gefunden.', true); } runNttBtn.disabled = false; }
